    res.text('PASSED');

    it('should work', function () {
        ai.mockIntent('intentName');


        const t = new Tester(bot);


        return t.text('Any text')
            .then(() => {
                t.actionPassed('intentAction');


            t.any()
                .contains('PASSED');
        })
    });
});
